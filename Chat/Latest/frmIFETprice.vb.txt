        txtIFETLian.Focus()

    Private Sub btnIFETOK_Click(sender As Object, e As EventArgs) Handles btnIFETOK.Click
        Dim IFETXondr As Decimal = 0
        Dim IFETLian As Decimal = CType(txtIFETLian.Text, Decimal)

        MsgBox(rowIndex & " " & ExchangesGivenOrTaken)

        IFETXondr = IFETLian / ((CType(txtIFETProfit.Text, Decimal) / 100) + 1) / ((CType(txtIFETFPA.Text, Decimal) / 100) + 1)

        If ExchangesGivenOrTaken = "given" Then
            frmCustomers.dgvGivenTo.Rows(rowIndex).Cells(3).Value = Qnt * IFETXondr ' Σύνολο (Ποσότητα x Χονδρική)
        ElseIf ExchangesGivenOrTaken = "taken" Then
            frmCustomers.dgvTakenFrom.Rows(rowIndex).Cells(3).Value = Qnt * IFETXondr ' Σύνολο (Ποσότητα x Χονδρική)
        End If

        Me.Close()
    End Sub
End Class
