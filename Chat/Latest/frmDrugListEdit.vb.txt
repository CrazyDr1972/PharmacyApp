    Private Sub UpdateDrugList(ByVal i As Integer)
        Dim insertData As String = ""

        Using con As SqlConnection = New System.Data.SqlClient.SqlConnection(connectionstring)
            con.Open()

            Dim chkRE As Boolean = False
            Dim chkRC As Boolean = False

            dle_description = EditedFormatedValue(i, 0)
            dle_morfi = EditedFormatedValue(i, 1)
            dle_lianiki = If(EditedFormatedValue(i, 3) = "", 0, CType(EditedFormatedValue(i, 3), Decimal))
            dle_xondriki = If(EditedFormatedValue(i, 2) = "", 0, CType(EditedFormatedValue(i, 2), Decimal))


            Try
                chkRE = CheckIfRecordExists(i)
                If chkRE = True Then
                    chkRC = CheckIfRecordChanged(i)
                End If
            Catch ex As Exception
            End Try


            ' Αν το προιον υπάρχει ήδη στο αρχείο και άλλαξε
            If chkRE = True AndAlso chkRC = True Then

                'MsgBox(" Line " & i.ToString & " exists!")
                insertData = "BEGIN TRANSACTION " & _
                                "UPDATE APOTIKH " & _
                                "SET APOTIKH.AP_DESCRIPTION = @Description, APOTIKH.AP_MORFI= @Morfi, APOTIKH.AP_TIMH_XON = @PriceXondr, APOTIKH.AP_TIMH_LIAN = @PriceLian, " & _
                                    "APOTIKH.AP_Free_1 = @Changed " & _
                                "WHERE APOTIKH.AP_ID = @Id " & _
                                 "COMMIT"
                '"UPDATE APOTIKH_BARCODES " & _
                '"SET APOTIKH_BARCODES.BRAP_AP_BARCODE = @Barcode " & _
                '"WHERE BRAP_AP_ID = @Id " & _


                ' Αν το προιον υπάρχει ήδη στο αρχείο και  ΔΕΝ άλλαξε
            ElseIf chkRE = True AndAlso chkRC = False Then
                insertData = ""

                ' αν δεν υπάρχει και το προσθέσαμε
            ElseIf chkRE = False Then
                Try
                    If dle_description = "" Then
                        MsgBox("Πρέπει να γράψεις περιγραφή προιόντος!")
                        Exit Sub
                    ElseIf dle_xondriki = 0 Then
                        MsgBox("Πρέπει να γράψεις χονδρική τιμή!")
                        Exit Sub
                    ElseIf dle_description <> "" And dle_xondriki > 0 Then


                        insertData = "BEGIN TRANSACTION " & _
                                        "INSERT INTO APOTIKH " & _
                                        "( APOTIKH.AP_ID, APOTIKH.AP_CODE, APOTIKH.AP_DESCRIPTION, APOTIKH.AP_MORFI, APOTIKH.AP_TIMH_XON, APOTIKH.AP_TIMH_LIAN, APOTIKH.AP_Free_1 ) " & _
                                        "VALUES (@APID, @Code, @Description, @Morfi, @PriceXondr, @PriceLian, @Changed) " & _
                                          "COMMIT"
                        '"INSERT INTO APOTIKH_BARCODES " & _
                        ' "(APOTIKH_BARCODES.BRAP_AP_BARCODE) " & _
                        ' "VALUES (@Barcode) " & _
                    End If
                Catch ex As Exception
                End Try


            End If

            If insertData <> "" Then
                Dim cmd As New SqlCommand(insertData, con)
                cmd.Parameters.AddWithValue("@Description", dle_description)
                cmd.Parameters.AddWithValue("@Morfi", dle_morfi)
                cmd.Parameters.AddWithValue("@PriceXondr", dle_xondriki)
                cmd.Parameters.AddWithValue("@PriceLian", dle_lianiki)
                cmd.Parameters.AddWithValue("@Changed", dle_changed)

                'cmd.Parameters.AddWithValue("@Barcode", EditedFormatedValue(i, 4))

                cmd.Parameters.AddWithValue("@Id", dgvEditDrugsList.Rows(i).Cells(5).Value)

                cmd.Parameters.AddWithValue("@APID", GetMaxIdFromDrugsList("select max(AP_ID) as myMax from apotikh") + 1)
                cmd.Parameters.AddWithValue("@Code", GetMaxIdFromDrugsList("select max(convert(int, AP_CODE)) as myMax from apotikh") + 1)

                Dim affRows As Integer = cmd.ExecuteNonQuery()

                If affRows > 0 Then MsgBox("(" & affRows.ToString & ") Records Updated !!")
            End If
        End Using

    End Sub

    Private Function EditedFormatedValue(ByVal i As Integer, ByVal cell As Integer) As String
        Dim result As String = ""

        result = dgvEditDrugsList.Rows(i).Cells(cell).GetEditedFormattedValue(i, DataGridViewDataErrorContexts.CurrentCellChange)
        result = If(IsDBNull(result), "", result)

        If (cell = 2 Or cell = 3) AndAlso result.Length > 3 AndAlso result.Substring(result.Length - 3, 2) = " €" Then
            result = result.Substring(0, result.Length - 2)
        ElseIf (cell = 2 Or cell = 3) And result = "" Then
            result = 0
        End If

        Return result

    End Function

    Private Function GetMaxIdFromDrugsList(ByVal sql As String) As Integer

        Using con As SqlConnection = New System.Data.SqlClient.SqlConnection(connectionString)
            con.Open()

            Dim cmd As New SqlCommand(sql, con)

            Dim myReader As SqlDataReader = cmd.ExecuteReader()

            If myReader.HasRows Then
                Do While myReader.Read()
                    Return myReader(0)
                Loop
            End If
        End Using
        Return 0
    End Function


    Private Function CheckIfRecordChanged(ByVal index As Integer) As Boolean
        Dim insertData As String = ""
        Dim changed = False

        Using con As SqlConnection = New System.Data.SqlClient.SqlConnection(connectionString)
            con.Open()

            insertData = "SELECT APOTIKH.AP_ID, APOTIKH.AP_CODE, APOTIKH.AP_DESCRIPTION, APOTIKH.AP_MORFI, APOTIKH.AP_TIMH_XON, APOTIKH.AP_TIMH_LIAN " & _
                          "FROM APOTIKH  " & _
                        "WHERE APOTIKH.AP_ID = @Id"

            Dim cmd As New SqlCommand(insertData, con)

            cmd.Parameters.AddWithValue("@Id", If(dgvEditDrugsList.Rows(index).Cells(5).Value, DBNull.Value))

            Dim myReader As SqlClient.SqlDataReader = cmd.ExecuteReader()

            If myReader.HasRows Then

                Do While myReader.Read()

                    ' Aν έχουν αλλάξει οι τιμές
                   
                    If If(IsDBNull(myReader("AP_DESCRIPTION")), "", myReader("AP_DESCRIPTION")) <> dle_description Then changed = True

                    If If(IsDBNull(myReader("AP_MORFI")), "", myReader("AP_MORFI")) <> dle_morfi Then changed = True

                    If If(IsDBNull(myReader("AP_TIMH_XON")), 0, myReader("AP_TIMH_XON")) <> dle_xondriki Then changed = True

                    If If(IsDBNull(myReader("AP_TIMH_LIAN")), 0, myReader("AP_TIMH_LIAN")) <> dle_lianiki Then changed = True

                    ' σημειώνει το record σαν changed
                    Dim IsTheRecordSignedAsChanged As String = If(IsDBNull(dgvEditDrugsList.Rows(index).Cells(7).Value), "", dgvEditDrugsList.Rows(index).Cells(7).Value)

                    If changed = True Then

                        dgvEditDrugsList.Rows(index).Cells(7).Value = "True"
                        dle_changed = True
                        Return True

                    End If

                Loop

            End If

            Return False

        End Using

    End Function




    Private Function CheckIfRecordExists(ByVal index As Integer) As Boolean
        Dim insertData As String = ""

        Using con As SqlConnection = New System.Data.SqlClient.SqlConnection(connectionString)
            con.Open()

            insertData = "SELECT * " & _
                        "FROM APOTIKH " & _
                        "WHERE APOTIKH.AP_ID = @Id"

            Dim cmd As New SqlCommand(insertData, con)

            cmd.Parameters.AddWithValue("@Id", If(dgvEditDrugsList.Rows(index).Cells(5).Value, DBNull.Value))

            Dim myReader As SqlClient.SqlDataReader = cmd.ExecuteReader()

            If myReader.HasRows Then

                Return True

            End If

            Return False

        End Using

    End Function



   



    Private Function DeleteCurrentDatagridRow() As Integer
        Dim insertData As String = ""
        Dim affRows As Integer = 0
        Dim id As Integer = dgvEditDrugsList.Rows(dgvEditDrugsList.SelectedRows(0).Index).Cells(5).Value

        Using con As SqlConnection = New System.Data.SqlClient.SqlConnection(connectionstring)
            con.Open()

            insertData = "DELETE " & _
                        "FROM APOTIKH " & _
                        "WHERE APOTIKH.AP_ID = @Id"

            Dim cmd As New SqlCommand(insertData, con)

            cmd.Parameters.AddWithValue("@Id", id)

            ' Επιβεβαίωση της διαγραφής
            If MessageBox.Show("Do you want to delete row # " & id & " ?", "Delete", MessageBoxButtons.YesNo) = DialogResult.Yes Then

                affRows = cmd.ExecuteNonQuery()
                dgvEditDrugsList.Rows.RemoveAt(dgvEditDrugsList.SelectedRows(0).Index)

            End If
        End Using

        Return affRows

    End Function

    Private Sub UpdateDrugIdInPricesAndExpirations()
        ' Μεταβλητές
        Dim sqlString As String = "UPDATE Expirations SET DrugId = {0} FROM Expirations WHERE Name = '{1}'"
        Dim sqlString2 As String = "UPDATE Prices SET DrugId = {0} FROM Prices WHERE Name = '{1}'"


        'Initialization νεας σύνδεσης με το connectionString που παίρνει από τις GlobalVariables
        ' στο οποίο πρέπει να προσθέσουμε το 'MultipleActiveResultSets' για να έχουμε nested SQLcommands
        Using con As New SqlClient.SqlConnection(connectionString & ";MultipleActiveResultSets=true")

            ' Ανοίγει την σύνδεση
            con.Open()

            'Creating two commands form current connection
            Dim cmd1 As SqlCommand = con.CreateCommand()
            Dim cmd2 As SqlCommand = con.CreateCommand()

            'Set the comment type
            cmd1.CommandType = CommandType.Text
            cmd2.CommandType = CommandType.Text

            'Setting the command text to first command
            cmd1.CommandText = "SELECT * FROM Drugs"

            'Execute the first command
            Dim idr1 As IDataReader = cmd1.ExecuteReader()

            While idr1.Read()

                'Διαβάζει το όνομα από το πρώτο command και ελέγχει ότι δεν είναι κενό
                If idr1("Name").ToString <> "" Then
                    ' Παίρνει το Id και το όνομα του φαρμάκου
                    Dim intDrugId As Integer = idr1("Id")
                    Dim strDrugName As String = idr1("Name")

                    ' αντικαθιστά το ' με '' για να αποφύγει λάθη από το SQL Transact
                    strDrugName = strDrugName.Replace("'", "''")

                    'create second command, που ψάχνει τα Expirations
                    cmd2.CommandText = String.Format(sqlString, intDrugId, strDrugName)

                    'Execute the reader
                    Dim idr2 As IDataReader = cmd2.ExecuteReader()

                    'Dont forgot to close second reader, this will just close reader not connection
                    idr2.Close()

                    'create third command, που ψάχνει τα Prices
                    cmd2.CommandText = String.Format(sqlString2, intDrugId, strDrugName)

                    'Execute the reader
                    idr2 = cmd2.ExecuteReader()

                    ' Δείκτης θέσης αρχείου
                    btnUpdateDrugName.Text = intDrugId.ToString
                    btnUpdateDrugName.Refresh()

                    'Dont forgot to close second reader, this will just close reader not connection
                    idr2.Close()

                End If


            End While

        End Using
    End Sub
