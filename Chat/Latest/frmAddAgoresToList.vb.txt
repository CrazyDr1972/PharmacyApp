    Private Sub GetSuppliersList()

        stringDTG = "SELECT distinct [Description] " & _
                       "FROM PharmacyCustomFiles.dbo.FarmAgores " & _

                       "ORDER BY [Description]"

        myTot = FillDatagrid(dgvSupliers, bsSuppliers, {"Προμηθευτής"}, {320}, {"0"}, {})

        ' Ενημερωτικό μύνημα
        Select Case myTot
            Case 0

                rtxtSuppliersMessage.Text = "Δεν βρέθηκαν προμηθευτές"

            Case 1

                rtxtSuppliersMessage.Text = "Βρέθηκε 1 προμηθευτής"

                ' Αλλάζει με κόκκινο χρώμα τον αριθμό
                HightlightInRichTextBox(rtxtSuppliersMessage, {"1"})

            Case Is > 1

                rtxtSuppliersMessage.Text = "Βρέθηκαν " & myTot.ToString & " προμηθευτές"

                ' Αλλάζει με κόκκινο χρώμα τον αριθμό
                HightlightInRichTextBox(rtxtSuppliersMessage, {myTot.ToString})

        End Select



    End Sub
