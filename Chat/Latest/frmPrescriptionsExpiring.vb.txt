    Private Sub UpdateProcessedDate()
        Dim insertData As String = ""
        Dim ChangedOrExists As String = ""
        Dim id As Integer = dgvPrescriptionsExpired.SelectedRows(0).Cells(5).Value

        Using con As SqlConnection = New System.Data.SqlClient.SqlConnection(connectionstring)
            con.Open()

            insertData = "UPDATE PharmacyCustomFiles.dbo.Prescriptions " & _
                          "SET [ProcessedDate]= @ProcessedDate " & _
                          "WHERE Id = @Id"



            For t = 0 To dgvPrescriptionsExpired.SelectedRows.Count - 1
                Dim cmd As New SqlCommand(insertData, con)

                With dgvPrescriptionsExpired
                    If chkDefaultDateProcessed.Checked = False Then
                        cmd.Parameters.AddWithValue("@ProcessedDate", dtpProcessDate.Value)
                    Else
                        cmd.Parameters.AddWithValue("@ProcessedDate", CType(dgvPrescriptionsExpired.SelectedRows(t).Cells(2).Value, Date))
                    End If

                    cmd.Parameters.AddWithValue("@Id", .SelectedRows(t).Cells(5).Value)
                End With

                cmd.ExecuteNonQuery()
            Next



                ''Ανανεώνει το Last Update
            frmCustomers.DisplayLastUpdate()

        End Using

    End Sub
