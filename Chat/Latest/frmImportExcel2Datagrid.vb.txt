    Private Function CheckIfParadrugCodeExists(ByVal code As String) As Integer
        Dim insertData As String = ""
        Dim AP_ID As Integer = 0

        Using con As SqlConnection = New System.Data.SqlClient.SqlConnection(connectionstring)
            con.Open()

            insertData = "SELECT AP_ID " & _
                        "FROM [Pharmacy2013C].[dbo].[APOTIKH] " & _
                        "WHERE AP_CODE = @Code"

            Dim cmd As New SqlCommand(insertData, con)

            cmd.Parameters.AddWithValue("@Code", code)

            Dim myReader As SqlClient.SqlDataReader = cmd.ExecuteReader()

            If myReader.HasRows Then
                Do While myReader.Read()
                    AP_ID = myReader(0)
                Loop

            End If

            Return AP_ID

        End Using

    End Function



    Private Function CheckIfBarcodeExists(ByVal barcode As String) As Integer
        Dim insertData As String = ""
        Dim AP_ID As Integer = 0

        Using con As SqlConnection = New System.Data.SqlClient.SqlConnection(connectionstring)
            con.Open()

            insertData = "SELECT * " & _
                        "FROM [Pharmacy2013C].[dbo].[APOTIKH_BARCODES] " & _
                        "WHERE BRAP_AP_BARCODE = @Barcode"

            Dim cmd As New SqlCommand(insertData, con)

            cmd.Parameters.AddWithValue("@Barcode", barcode)

            Dim myReader As SqlClient.SqlDataReader = cmd.ExecuteReader()

            If myReader.HasRows Then
                Do While myReader.Read()
                    AP_ID = myReader(1)
                Loop

            End If

            Return AP_ID

        End Using

    End Function

    Private Sub UpdateBarcodesWithParadrugs(ByVal AP_ID As Integer, barcode As String, mode As String)
        Dim insertData As String = ""
        Dim DrugFPA As Integer = 0

        Using con As SqlConnection = New System.Data.SqlClient.SqlConnection(connectionstring)
            con.Open()
            If mode = "UpdateByID" Then
                insertData = "UPDATE Pharmacy2013C.dbo.APOTIKH_BARCODES " & _
                            "SET [BRAP_AP_BARCODE] = @barcode " & _
                            "WHERE AP_ID = @Id"
            ElseIf mode = "UpdateByBarcode" Then
                insertData = "UPDATE Pharmacy2013C.dbo.APOTIKH_BARCODES " & _
                            "SET [BRAP_AP_ID] =  @AP_ID  " & _
                            "WHERE BRAP_AP_BARCODE = @barcode"
            Else
                insertData = "INSERT INTO Pharmacy2013C.dbo.APOTIKH_BARCODES " & _
                                                     "([BRAP_AP_BARCODE] ,[BRAP_AP_ID]) " & _
                                                     "VALUES (@barcode, @AP_Id)"
            End If


            Dim cmd As New SqlCommand(insertData, con)

            ' Περνάει τις παραμέτρους του SQL 
            If mode = "UpdateByID" Then
                cmd.Parameters.AddWithValue("@AP_Id", AP_ID)
            ElseIf mode = "UpdateByBarcode" Then
                cmd.Parameters.AddWithValue("@barcode", barcode)
            Else
                cmd.Parameters.AddWithValue("@AP_Id", AP_ID)
                cmd.Parameters.AddWithValue("@barcode", barcode)
            End If


            ' Σώζει τις αλλαγές
            Try
                cmd.ExecuteNonQuery()
            Catch ex As Exception
                'MsgBox("ERROR (Barcode)!!!" & vbCrLf & ex.Message)
            End Try

        End Using

    End Sub


    Private Function UpdateApotikiWithParadrugs(ByVal AP_Id As Integer, code As String, fullname As String, lian As Decimal, xondr As Decimal, notes As String, fpa As Decimal) As Integer
        Dim insertData As String = ""
        Dim DrugFPA As Integer = 0
        Dim new_AP_ID As Integer = 0
        Dim ApotikiLastRecord As Integer = 0

        Using con As SqlConnection = New System.Data.SqlClient.SqlConnection(connectionstring)
            con.Open()

            If AP_Id > 0 Then
                insertData = "UPDATE Pharmacy2013C.dbo.APOTIKH " & _
                             "SET [AP_FAR_ID] = @AP_FAR_ID, [AP_DESCRIPTION] = @AP_DESCRIPTION, [AP_KA_ID] =  @AP_KA_ID, [AP_YPK_ID] =  @AP_YPK_ID, [AP_MEMO_PWL_CUSTOM] = @Notes, " & _
                             "[AP_FP_ID] = @AP_FP_ID, [AP_TIMH_XON] =  @AP_TIMH_XON,[AP_TIMH_LIAN] = @AP_TIMH_LIAN, [AP_CHANGE_DATE] =  @AP_CHANGE_DATE,  [AP_ACCEPT_TIMH] =  @AP_ACCEPT_TIMH  " & _
                             "WHERE AP_ID = @Id"
            ElseIf AP_Id = 0 Then
                insertData = "INSERT INTO Pharmacy2013C.dbo.APOTIKH " & _
                            "([AP_ID], [AP_FAR_ID] ,[AP_CODE] ,[AP_DESCRIPTION] ,[AP_KA_ID] ,[AP_YPK_ID], [AP_MEMO_PWL_CUSTOM], [AP_FP_ID], [AP_TIMH_XON], [AP_TIMH_LIAN], [AP_CREATE_DATE], [AP_ACCEPT_TIMH]) " & _
                            "VALUES (@AP_Id, @AP_FAR_ID, @AP_CODE, @AP_DESCRIPTION, @AP_KA_ID, @AP_YPK_ID, @Notes, @AP_FP_ID, @AP_TIMH_XON, @AP_TIMH_LIAN, @AP_CREATE_DATE, @AP_ACCEPT_TIMH)"
            End If

            Dim cmd As New SqlCommand(insertData, con)

            ' Περνάει τις παραμέτρους του SQL 
            Select Case fpa
                Case Is = 0.24
                    DrugFPA = 2
                Case Is = 0.13
                    DrugFPA = 3
                Case Is = 0.06
                    DrugFPA = 1
            End Select

            cmd.Parameters.AddWithValue("@AP_FAR_ID", 1)
            cmd.Parameters.AddWithValue("@Id", AP_Id)
            cmd.Parameters.AddWithValue("@AP_DESCRIPTION", fullname)
            cmd.Parameters.AddWithValue("@AP_CODE", code)
            cmd.Parameters.AddWithValue("@AP_KA_ID", 13)
            cmd.Parameters.AddWithValue("@AP_OM_ID", 2)
            cmd.Parameters.AddWithValue("@AP_YPK_ID", 35)
            cmd.Parameters.AddWithValue("@AP_FP_ID", DrugFPA)
            cmd.Parameters.AddWithValue("@AP_TIMH_XON", xondr)
            cmd.Parameters.AddWithValue("@AP_TIMH_LIAN", lian)
            cmd.Parameters.AddWithValue("@AP_ACCEPT_TIMH", 0)
            cmd.Parameters.AddWithValue("@Notes", If(notes Is Nothing, DBNull.Value, notes))

            If AP_Id > 0 Then
                cmd.Parameters.AddWithValue("@AP_CHANGE_DATE", Now())
                new_AP_ID = AP_Id
            ElseIf AP_Id = 0 Then
                ApotikiLastRecord = ExecuteSQLTransact("Select max([AP_Id]) FROM [Pharmacy2013C].[dbo].[APOTIKH]")
                new_AP_ID = ApotikiLastRecord + 1
                cmd.Parameters.AddWithValue("@AP_Id", new_AP_ID)
                cmd.Parameters.AddWithValue("@AP_CREATE_DATE", Now())

            End If

            ' Σώζει τις αλλαγές
            Try
                cmd.ExecuteNonQuery()
            Catch ex As Exception
                If ex.Message.Contains("String or binary data would be truncated.") Then
                    errorTruncated += 1
                    lblInfo2.Text = "Entries truncated: " & errorTruncated
                Else
                    MsgBox("ERROR (APOTIKI) !!!" & vbCrLf & ex.Message)
                End If

            End Try

        End Using
        Return new_AP_ID
    End Function
