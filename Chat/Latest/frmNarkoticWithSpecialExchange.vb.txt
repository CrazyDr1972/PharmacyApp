    Private Sub btnSpecialExchangeYes_Click(sender As Object, e As EventArgs) Handles btnSpecialExchangeYes.Click
        If ExchangesGivenOrTaken = "given" Then
            frmCustomers.dgvGivenTo.SelectedRows(0).Cells(5).Value = "X"
            frmCustomers.dgvGivenTo.CurrentCell = frmCustomers.dgvGivenTo.SelectedRows(0).Cells(4)
        ElseIf ExchangesGivenOrTaken = "taken" Then
            frmCustomers.dgvTakenFrom.SelectedRows(0).Cells(5).Value = "X"
            frmCustomers.dgvTakenFrom.CurrentCell = frmCustomers.dgvTakenFrom.SelectedRows(0).Cells(4)
        End If

        Me.Close()
    End Sub

    Private Sub btnSpecialExchangeNo_Click(sender As Object, e As EventArgs) Handles btnSpecialExchangeNo.Click
        If ExchangesGivenOrTaken = "given" Then
            frmCustomers.dgvGivenTo.SelectedRows(0).Cells(5).Value = ""
            frmCustomers.dgvGivenTo.CurrentCell = frmCustomers.dgvGivenTo.SelectedRows(0).Cells(4)

        ElseIf ExchangesGivenOrTaken = "taken" Then
            frmCustomers.dgvTakenFrom.SelectedRows(0).Cells(5).Value = ""
            frmCustomers.dgvTakenFrom.CurrentCell = frmCustomers.dgvTakenFrom.SelectedRows(0).Cells(4)
        End If

        Me.Close()
    End Sub
