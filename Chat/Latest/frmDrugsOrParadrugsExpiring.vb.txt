    Private Function DisplayCustomDatagrid_DrugsExpiring(ByVal mysql As String) As Integer

        'Initialization νεας σύνδεσης με το connectionString που παίρνει από τις GlobalVariables
        con = New SqlConnection(connectionstring)

        'που την ανοίγει εδώ
        con.Open()

        'Initialization νέου CommandAdapter με την SqlString που παίρνει σαν εξωτερική παράμετρο
        cmdDTG = New SqlCommand(mysql, con)
        daDTG = New SqlDataAdapter(cmdDTG)  'Ορισμός νέου SqlDataAdapter και νέου DataSet
        cbDTG = New SqlCommandBuilder(daDTG)
        dsDTG = New DataSet

        ' και ο SqlDataAdapter γεμίζει το Dataset
        daDTG.Fill(dsDTG, "DTG-DrExp")
        ' το οποίο γεμίζει το datatable
        dtDTG = dsDTG.Tables("DTG-DrExp")

        'Καθορίζει το source του BindingSource ως το Datatable
        bsExpiringProducts.DataSource = dtDTG

        'Κλείνει την σύνδεση
        con.Close()


        With dgvExpiringProducts

            'Αδειάζει το datagridView
            .Columns.Clear()

            'Εμποδίσει το Datagrid να εμφανίσει αυτόματα τα Columns
            .AutoGenerateColumns = False

            'Καθορίζει το  source του DataGrid ως το BindingSource
            .DataSource = bsExpiringProducts

        End With

        'Όρίζει το 1ο πεδίο του Datagrid 
        Dim Product As New DataGridViewTextBoxColumn
        With Product
            ' και του δίνει τη τιμή του αντίστοιχου πεδίου
            .DataPropertyName = "ProductName"
            ' Formatting..
            .HeaderText = "Προιόν"
            .Width = 246
            .DefaultCellStyle.Format = ""
        End With

        'Όρίζει το 2ο πεδίο του Datagrid 
        Dim ExpDate As New DataGridViewTextBoxColumn
        With ExpDate
            ' και του δίνει τη τιμή του αντίστοιχου πεδίου
            .DataPropertyName = "ExpDate"
            ' Formatting..
            .HeaderText = "Λήξη"
            .Width = 60
            .DefaultCellStyle.Format = ""
        End With

        Dim Id As New DataGridViewTextBoxColumn
        With Id
            ' και του δίνει τη τιμή του αντίστοιχου πεδίου
            .DataPropertyName = "Id"
            ' Formatting..
            .HeaderText = "Id"
            .Width = 60
            .DefaultCellStyle.Format = ""
        End With

        Dim ComplExpDate As New DataGridViewTextBoxColumn
        With ComplExpDate
            ' και του δίνει τη τιμή του αντίστοιχου πεδίου
            .DataPropertyName = "CompleteExpDate"
            ' Formatting..
            .HeaderText = "CompleteExpDate"
            .Width = 220
            .DefaultCellStyle.Format = ""
        End With

        With dgvExpiringProducts
            'Εμφανίζει τα columns του Datagrid
            .Columns.Add(Product)
            .Columns.Add(ExpDate)
            .Columns.Add(Id)
            .Columns.Add(ComplExpDate)

            ' Εναλλαγή του χρωματισμού των rows
            .RowsDefaultCellStyle.BackColor = Color.Bisque
            .AlternatingRowsDefaultCellStyle.BackColor = Color.Beige

            'Εξαφανίζει τo πεδίο "Id"
            .Columns(2).Visible = False
            .Columns(3).Visible = False
        End With

        'Επιστρέφει τον συνολικό αριθμό εγγραφών του Datagrid
        Return dsDTG.Tables("DTG-DrExp").Rows.Count

    End Function



    Private Sub DeleteExpiration()
        Dim insertData As String = ""
        Dim id As Integer = 0

        Try
            id = dgvExpiringProducts.SelectedRows(0).Cells(2).Value
        Catch ex As Exception
        End Try

        Using con As SqlConnection = New System.Data.SqlClient.SqlConnection(connectionstring)
            con.Open()

            insertData = "DELETE FROM PharmacyCustomFiles.dbo.Expirations " & _
                         "WHERE Id = @Id"

            ' Επιβεβαίωση της διαγραφής
            If MessageBox.Show("Θέλετε να διαγράψετε τη λήξη # " & id & " ?", "Διαγραφή", MessageBoxButtons.YesNo) = DialogResult.Yes Then

                ' Aφαίρεση της επιλεγμένης εγγραφής από το DataGRidView
                'dgvExchangesFrom.Rows.Remove(dgvExchangesFrom.Rows(selectedRow))

                Dim cmd As New SqlCommand(insertData, con)
                cmd.Parameters.AddWithValue("@Id", id)

                cmd.ExecuteNonQuery()
            End If
        End Using

    End Sub
